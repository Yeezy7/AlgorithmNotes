int bin_search(int *a, int n, int x){   //在数组a中找数字x，返回位置
    int left = 0, right = n;
    while (left < right) {
        int mid = left+(right-left)/2;
        if (a[mid] >= x)  right = mid;
        else    left = mid + 1;
        cout << a[mid]<<"   ";   //调试：打印出猜数的过程
    }
   return left;
