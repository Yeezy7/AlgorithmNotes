## 原地归并两个升序链表

迭代法 + 虚拟头节点：空间复杂度为 O(1)

```c++
/**
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 *     ListNode(int x) : val(x), next(nullptr) {}
 * };
 */
class Solution {
public:
    ListNode* Merge(ListNode* pHead1, ListNode* pHead2) {
        // 1. 创建虚拟头节点（辅助节点，不存实际数据）
        ListNode* dummy = new ListNode(0);
        ListNode* cur = dummy;  // cur 始终指向新链表的尾节点

        // 2. 同时遍历两个链表，比较节点值
        while (pHead1 != nullptr && pHead2 != nullptr) {
            if (pHead1->val <= pHead2->val) {
                cur->next = pHead1;   // 将 pHead1 接到新链表尾部
                pHead1 = pHead1->next;
            } else {
                cur->next = pHead2;
                pHead2 = pHead2->next;
            }
            cur = cur->next;          // 移动尾指针
        }

        // 3. 将剩余未合并的部分直接接上
        cur->next = (pHead1 == nullptr) ? pHead2 : pHead1;

        // 4. 返回合并后的链表头（跳过虚拟节点）
        return dummy->next;
    }
};
```
