void Merge(long long l, long long mid, long long r) {
    long long i = l, j = mid+1, t = 0;
    while(i <= mid && j <= r) {
        if(a[i] > a[j]) {
            b[t++] = a[j++];
        }
        else
            b[t++] = a[i++];
    }
    while(i <= mid)
        b[t++] = a[i++];
    while(j <= r)
        b[t++] = a[j++];
    for(i=0;i<t;++i)
        a[l+i] = b[i];
}

void MergeSort(long long l, long long r) {
    if(l < r) {
        long long mid = (l + r) / 2;
        MergeSort(l, mid);
        MergeSort(mid+1, r);
        Merge(l, mid, r);
    }
}
