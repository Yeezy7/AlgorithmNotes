## 1、保留精度

## 6、大数取余

思想：

1. 转成字符串

2. 从高位累加（乘10再加当前位置的数）

3. 取余并替换

示例：

```
#include <iostream>
#include <string>
using namespace std;
 
int main() {
    string num = '1234562324242424242424';   // 被除数
    int divider = 300;   // 除数
    int total = 0;
    for(int i=0;i<num.size();++i){
        int digit = num[i] - '0';
        total *= 10;
        total += digit;
        total %= divider;
    }
    cout << total;   // 输出结果为余数
}
```

## 9、进制转换

- 10进制转x进制代码

```
char intTochar(int num){
    if(num >= 0 && num <= 9){
        return num+'0';
    }
    return num-10+'A';
}
// 10 -> b
string decTob(long dec_num, int b){
    string str = "";
    while(dec_num != 0){
        str.append(string(1, intTochar(dec_num % b)));
        dec_num /= b;
    }
    reverse(str.begin(),str.end()); // 翻转
    return str;
}
```

- x进制转10进制

```
int charToint(char c){
    if(c>='A' && c <= 'Z'){
        return c-'A'+10;
    }
    else if(c>='a' && c <= 'z'){
        return c-'a'+10;
    }
    return c-'0';
}
// a -> 10
long aTodec(int a, string str){
    long res = 0;
    for(int i=0;i<str.size();++i){
        res *= a;
        res += charToint(str[i]);
    }
    return res;
}
```

## 10、同余模定理及大数取模

### 10.1 同余模定理

- (a+b)%c=(a%c+b%c)%c

- (a*a*a)%n=((a%n)*(a%n)*(a%n))%n

示例：

```
#include<stdio.h>  
#include<iostream>  
using namespace std;  
  
int main() {  
    long long int n;  
    while(scanf("%lld",&n) != EOF) {  
        long long int s=n;  
        for(int i=1;i<5;i++) {  
            s=((s%3)*(n%3));  // 同余模定理
        }  
        printf("%lld\n",s%3);  // 最后再取余
    }  
    return 0;  
}  
```

### 10.2 大数取模

联想到进制转换时的方法；

设大数 m=1234, 模 n，就等于 ((((1 * 10) % n + 2 % n) % n * 10 % n + 3 % n) % n * 10 % n + 4 % n)%n

示例：

```
#include <stdio.h>  
  
char s[1000];  
int main() {  
    int i, j, k, m, n;   // m大数,n模
    while(scanf("%s%d", s, &n) != EOF) {  
        m = 0;  
        for(i = 0; s[i] != '\0'; i++)  
            m = ((m * 10) % n + (s[i] - '0') % n) % n;   上个余数*10+当前数再取余
        printf("%d\n", m);  
    }  
    return 0;  
}  
```

## 11、最大公约数及最小公倍数

### 11.1 最大公约数(GCD)

```
#include <stdio.h>  
// 记住下面套路：
int gcd(int a, int b) {  
    if (b == 0) return a;  
    else return gcd(b, a % b);  
}  
int main() {  
    int x, y;  
    scanf("%d%d", &x, &y);  
    printf("%d\n", gcd(x, y));  
} 
```

### 11.2 最小公倍数(LCM)

LCM(x, y) = x * y / GCD(x, y)

即：**两个数的最小公倍数等于 两个数的乘积 除以 两个数的最大公约数**。

根据上述方法很容易得到最大公约数。

再度变形可得到

x * y = LCM(x, y) * GCD(x, y)

即：通过最小公倍数*最大公约数可得到x*y

```
int quick_pow(int num, int pow){
    int res = 1;
    while(pow > 0){
        if(pow % 2 == 1){
            res *= num;
        }
        pow /= 2;
        num *= num;
    }
```

## 15、大数进制转换

将一个长度最多为30位数字的十进制非负整数转换为二进制数输出

```
#include <iostream>
#include <bits/stdc++.h>
using namespace std;

char a[31];
int num[31];
int res[31];

bool iszero(int n) {
    for (int i = 0; i < n; ++i) {
        if (num[i] != 0) {
            return true;
        }
    }
    return false;
}

int main() {
    while (cin >> a) {
        
        int len = strlen(a);
        for (int i = 0; i < len; ++i) {
            num[i] = a[i] - '0';
        }
        if (!iszero(len)) {
            cout << 0 << endl;
            continue;
        }
        int pos = 0;
        while(iszero(len)){
            int remain = 0;
            for (int i = 0; i < len; ++i) {
                int temp = remain * 10 + num[i];
                num[i] = temp / 2;
                remain = temp % 2;
            }
            res[pos++] = remain;
        }

        for (int i = pos - 1; i >= 0; --i) {
            cout << res[i];
        }
        cout << endl;
    }


    return 0;
}
```

## 16、Kruskal模板

题目：[1312 - 畅通工程 _N诺计算机考研 (noobdream.com)](https://noobdream.com/DreamJudge/Issue/page/1312/)

```
#include <iostream>
#include <algorithm>
using namespace std;

struct Edge {
    int a, b;
    int val;

};
Edge edge[101];

bool cmp(Edge l, Edge r) {
    return l.val < r.val;
}

// 并查集
int father[101];
int find(int x) {  
    if (father[x] == x) {
        return x;
    }
    father[x] = find(father[x]); // 路径压缩
    return father[x];
}

void init(int n) {
    for (int i = 1; i <= n; ++i) {
        father[i] = i;
    }
}

int main() {

    int n, m;
    while (cin >> n >> m) {
        if (n == 0) {
            break;
        }
        init(n);
        for (int i = 0; i < n; ++i) {
            cin >> edge[i].a >> edge[i].b >> edge[i].val;
        }
        sort(edge, edge + n, cmp);

        int res = 0;
        int count = 0;
        for (int i = 0; i < n; ++i) {
            int a = edge[i].a;
            int b = edge[i].b;
            a = find(a);
            b = find(b);
            if (a != b) {
                res += edge[i].val;
                father[a] = b;
                count++;
            }
        }
        if (count == (m - 1)) {
            cout << res << endl;
        }
        else {
            cout << "?" << endl;
        }

    }

    return 0;
}
```
